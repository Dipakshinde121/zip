8. FORM VALIDATION (HTML + JAVASCRIPT)
====================================================
<!DOCTYPE html>
<html>
<head>
<title>Validation</title>
<script>
function validateForm(){
    var name=document.getElementById("name").value;
    if(name==""){
        alert("Name is required");
        return false;
    }
    alert("Form Submitted Successfully");
    return true;
}
</script>
</head>
<body>
<form onsubmit="return validateForm()">
Name:<br>
<input type="text" id="name" required><br><br>

Email:<br>
<input type="email" required><br><br>

Password:<br>
<input type="password" minlength="6" required><br><br>

<input type="submit">
</form>
</body>
</html>

9. PHOTO GALLERY
====================================================
<!DOCTYPE html>
<html>
<head>
<style>
.large{width:400px;height:250px;border:2px solid black;}
.small{width:100px;height:70px;margin:5px;cursor:pointer;}
</style>
<script>
function changeImage(img){
document.getElementById("big").src=img;
}
</script>
</head>
<body>

<h2>Photo Gallery</h2>

<img id="big" src="1.jpg" class="large"><br><br>

<img src="1.jpg" class="small" onclick="changeImage('1.jpg')">
<img src="2.jpg" class="small" onclick="changeImage('2.jpg')">
<img src="3.jpg" class="small" onclick="changeImage('3.jpg')">
<img src="4.jpg" class="small" onclick="changeImage('4.jpg')">

</body>
</html>
